        let selectedRole = null;
        
        const roleConfig = {
            reception: {
                name: 'Reception',
                label: 'RECEPTION/CASHIER',
                welcome: 'Welcome back, Jane!',
                subtitle: "Here's what's happening with your patients today."
            },
            admin: {
                name: 'Admin',
                label: 'ADMINISTRATION',
                welcome: 'Admin Dashboard',
                subtitle: 'Manage your healthcare facility operations.'
            },
            lab: {
                name: 'Laboratory',
                label: 'LABORATORY',
                welcome: 'Laboratory Dashboard',
                subtitle: 'Manage lab tests and results.'
            },
            consultation: {
                name: 'Consultation',
                label: 'MEDICAL',
                welcome: 'Consultation Dashboard',
                subtitle: 'Manage patient consultations and medical records.'
            },
            pharmacy: {
                name: 'Pharmacy',
                label: 'PHARMACY',
                welcome: 'Pharmacy Dashboard',
                subtitle: 'Manage prescriptions and medication inventory.'
            }
        };

        function selectRole(role) {
            // Remove selected class from all cards
            document.querySelectorAll('.role-card').forEach(card => {
                card.classList.remove('selected');
            });
            
            // Add selected class to clicked card
            event.currentTarget.classList.add('selected');
            
            selectedRole = role;
            document.getElementById('loginBtn').disabled = false;
            document.getElementById('loginBtn').textContent = `Continue as ${roleConfig[role].name}`;
        }

        function login() {
            if (!selectedRole) return;
            
            // Hide login screen
            document.getElementById('loginScreen').style.display = 'none';
            
            // Show dashboard
            const dashboard = document.getElementById('dashboardScreen');
            dashboard.classList.add('active');
            dashboard.setAttribute('data-current-role', selectedRole);
            
            // Update UI with role-specific content
            const config = roleConfig[selectedRole];
            document.getElementById('navLabel').textContent = config.label;
            document.getElementById('userRoleText').textContent = config.name;
            document.getElementById('welcomeTitle').textContent = config.welcome;
            document.getElementById('welcomeSubtitle').textContent = config.subtitle;
        }

        function logout() {
            // Reset selection
            selectedRole = null;
            document.querySelectorAll('.role-card').forEach(card => {
                card.classList.remove('selected');
            });
            document.getElementById('loginBtn').disabled = true;
            document.getElementById('loginBtn').textContent = 'Select a Role to Continue';
            
            // Hide dashboard
            document.getElementById('dashboardScreen').classList.remove('active');
            
            // Show login screen
            document.getElementById('loginScreen').style.display = 'flex';
        }
    </script>
